36  	 * Explode an EAR. In general, earFile should be an archive with the "ear" extension. However, it is legitimate to
37  	 * explode directories (that may or may not have the EAR extension) if they contain un-exploded WARs inside.
38  	 *
39  	 * @param earFile
40  	 * @return
41  	 */
42  	public boolean explodeEarFile(File earFile) {
43  		if (earFile.exists()) {
44  			if (!earFile.getName().endsWith("ear") && !earFile.isDirectory()) {
45  				log.warn("Attempting to explode a non-directory file that does not have the 'ear' extension.");
46  			}
47  			try {
48  				FileHelper.explodeWebApp(earFile);
49  
50  				if (!earFile.isDirectory()) {
51  					log.error("Could not explode EAR file for unknown reasons. Please explode it manually and invoke this goal again.");
52  				} else if (!FileHelper.containsGeneratedSources(earFile)) {
53  					log.warn("Exploded EAR does not contain generated Java sources.");
54  				}
55  				return true;
56  			} catch (IOException e) {
57  				log.error("Error while exploding EAR file located at " + earFile.getAbsolutePath(), e);
58  			}
59  		}
60  		return false;
61  	}

63  	public static final String XML_PATH_SEP = "/";
64  	/**
65  	 * Simple way to access a certain node in an XML document. Not meant to replace XPath at all.
66  	 * 
67  	 * @param rootNode  Must be non-null. Starting point.
68  	 * @param path      Must be non-null. A path delimited by /.
69  	 * @return          At worst, an empty array. Otherwise, the node at the path specified xor the children if children
70  	 *                  exist.
71  	 */
72  
73  	public static Xpp3Dom[] getElem(Xpp3Dom rootNode, String path) {
74  		assert rootNode != null;
75  		assert path != null;
76  
77  		Xpp3Dom[] emptyArray = new Xpp3Dom[] { };
78  
79  		if (rootNode == null || !StringHelper.isDefinedNonTrivially(path)) {
80  			return emptyArray;
81  		}
82  
83  		String[] pathElems = path.split(XML_PATH_SEP);
84  		Xpp3Dom currNode = rootNode;
85  		for (int i = 0; i < pathElems.length; i++) {
86  			if (!StringHelper.isDefinedNonTrivially(pathElems[i])) { // Don't attempt to get a child without a name.
87  				continue;
88  			}
89  			currNode = currNode.getChild(pathElems[i]);
90  			if (currNode == null) {
91  				return emptyArray; // We cannot continue. This path does not exist.
92  			}
93  		}
94  
95  		// Scenario 1: This node *can* and *does* have children. Return its children in an array.
96  		if (currNode.getChildCount() > 0) {
97  			return currNode.getChildren();
98  		}
99  
100 		// Scenario 2: This node isn't the kind that has children. Rather, it takes a value. Just return the node.
101 		// If desired, the user may query for the value him/herself.
102 
103 		if (StringHelper.isDefinedNonTrivially(currNode.getValue())) {
104 			return new Xpp3Dom[] { currNode };
105 		}
106 
107 		// Scenario 3: Since this node doesn't have a value, it is one that *can* have children but doesn't. Indicate
108 		// this to the user of this method by returning an empty array.
109 		return emptyArray;
110 	}

113 	 * Given a path to a node (relative to a specified node), return its value.
114 	 * 
115 	 * @param rootNode
116 	 * @param path
117 	 * @returns The value of the specified node. Null if the specified node does not exist or has siblings.
118 	 */
119 	public static String getElemValue(Xpp3Dom rootNode, String path) {
120 		Xpp3Dom[] elem = getElem(rootNode, path);
121 		String retVal = null;
122 		if (elem.length == 1) {
123 			retVal = elem[0].getValue();
124 		}
125 		return retVal;

145 	 * Returns the classpath string. The includeTests variable if true will
146 	 * result in Test class path dependencies to be included.
147 	 *
148 	 * @return An OS specific path string for passing to the -cp flag of SCA.
149 	 *         Returns null if none is found.
150 	 */
151 	public String resolveClassPath(List<Artifact> cpathArtifacts) {
152 		StringBuffer cpathBuff = new StringBuffer();
153 		Iterator<Artifact> artifactIt = cpathArtifacts.iterator();
154 		int size = cpathArtifacts.size();
155 		int count = 1;
156 
157 		log.debug("** Artifacts ** ");
158 
159 		while (artifactIt.hasNext()) {
160 			Artifact artifact = artifactIt.next();
161 			log.debug("#" + count + " scope: " + artifact.getScope() + " type:" + artifact.getType() + ", @" + artifact.getFile().getAbsolutePath());
162 			cpathBuff.append(artifact.getFile().getAbsolutePath());
163 
164 			count++;
165 
166 			// add separator if we're not the last
167 			if (count <= size) {
168 				cpathBuff.append(File.pathSeparator);
169 			}
170 		}
171 
172 		// return constructed string or null if empty
173 		if (cpathBuff.length() != 0) {
174 			return cpathBuff.toString();
175 		}else {
176 			return null;
177 		}
178 	}

186 	public Plugin getPlugin(String groupId, String artifactId) {
187 		List plugins = project.getBuildPlugins();
188 
189 		if (plugins != null) {
190 			Iterator iter = plugins.iterator();
191 			while (iter.hasNext()) {
192 				Plugin currPlugin = (Plugin)iter.next();
193 
194 				String currGroupId = currPlugin.getGroupId();
195 				if (!currGroupId.equals(groupId)) {
196 					continue;
197 				}
198 
199 				String currArtifactId = currPlugin.getArtifactId();
200 				if (!currArtifactId.equals(artifactId)) {
201 					continue;
202 				}
203 
204 				return currPlugin;
205 			}
206 		}
207 		return null;
208 	}

280 	 * This is an inner class of Util so that is is able to access the logging mechanism of a Mojo. Using an instance
281 	 * of this class, it is possible to get the output of a command invocation.

283 	public class Executable {
284 		private List outputLines = new ArrayList();
285 		private List errorLines = new ArrayList();
286 		private StreamConsumer output = new StreamConsumer() {
287 			public void consumeLine(String line) {
288 				outputLines.add(line);
289 			}
290 		};
291 
292 		private StreamConsumer error = new StreamConsumer() {
293 			public void consumeLine(String line) {
294 				errorLines.add(line);
295 			}
296 		};
297 
298 		private Commandline cli = new Commandline();
299 
300 		public Executable(String executableName) {
301 			cli.setExecutable(executableName);
302 			try {
303 				cli.addSystemEnvironment();
304 			} catch (Exception e) {
305 				log.error("Error setting environment variables: ", e);
306 			}
307 		}
308 
309 		/**
310 		 * Convenience constructor. If not executable is specified, assume that it is sourceanalyzer.
311 		 */
312 		public Executable() {
313 			this("sourceanalyzer");
314 		}
315 
316 		private void addOption(String arg) {
317 			try {
318 				cli.createArgument().setLine("\"" + arg + "\"");
319 			} catch (Exception e) {
320 				log.error("Could not add argument " + arg + ".");
321 			}
322 		}
323 
324 		public void addOptions(String... optionArr) {
325 			for (String opt : optionArr) {
326 				addOption(opt);
327 			}
328 		}

330 		public boolean invoke() {
331 			log.info("Executing command: " + cli.toString());
332 			int exitCode = 0;
333 			try {
334 				exitCode = CommandLineUtils.executeCommandLine(cli, output, error);
335 				if (exitCode != 0) {
336 					log.error("Command execution failed. Exit code: " + exitCode + ". Make sure " + "that the executable is available.");
337 				}
338 			} catch (CommandLineException e) {
339 			    log.error("Error executing command: ", e);
340 			}
341 
342 			return exitCode == 0;
343 		}

345 		/**
346 		 * Retrieves non-warning output from an invocation of the executable.
347 		 * @return
348 		 */
349 		public String getOutput() {
350 			Iterator it = outputLines.iterator();
351 			while (it.hasNext()) {
352 				String currElem = (String)it.next();
353 				if (currElem.startsWith("[warning]")) {
354 					it.remove();
355 				}
356 			}
357 			return StringHelper.listToString(outputLines, File.pathSeparator);

466     /**
467      * Invoke sourceanalyzer using an argument file.
468      * 
469      * @param argFile for SCA invocation.
470      * @param failOnSCAError If true, throw an exception if the command fails.
471      * @return True if the invocation succeeded. Otherwise, false.
472      * @throws org.apache.maven.plugin.MojoExecutionException
473      * @throws org.apache.maven.plugin.MojoFailureException
474      */
475     public boolean invoke(String sourceanalyzer, String args, String argFile, boolean failOnSCAError, String maxHeap)
476 		throws MojoExecutionException, MojoFailureException {
477 		
478 		StreamConsumer output = new DefaultConsumer();
479 		StreamConsumer error = new DefaultConsumer();
480 		Commandline cli = new Commandline();
481 
482 		try {
483 			if (sourceanalyzer == null) {
484 			    throw new IllegalArgumentException("No executable defined.");
485 			}
486 			cli.setExecutable(sourceanalyzer);
487 			cli.addSystemEnvironment();
488 
489 			// The JVM heap space must be specified/allocated prior to starting up the JVM. Hence, it cannot go inside
490 			// the argument file.
491 			if (maxHeap != null && maxHeap.length() > 0) {
492 			    cli.createArg().setLine(args);
493 			}
494 
495 			cli.createArg().setValue("@" + argFile);
496 
497 			log.info("Executing command: \"" + cli.toString() + "\"") ;
498 			int result = CommandLineUtils.executeCommandLine(cli, output, error);
499 			if (result != 0) {
500 				String errorMsg = "Error invoking sourceanalyzer. Exit code: " + result +
501 				        ".\nVerify your project settings and your SCA installation.";
502 				log.error(errorMsg);
503 				if (failOnSCAError) {
504 				    throw new MojoExecutionException(errorMsg);
505 				}
506 			}
507 			return result == 0;
508 		} catch (CommandLineException e) {
509 			log.error("Exception while executing command: ", e);
510 			log.error("Please confirm that the sourceanalyzer executable is in the path or specified correctly using the sourceanalyzer configuration" +
511 			    " option. Currently it is set to \"" + sourceanalyzer + "\"") ;
512 		    throw new MojoExecutionException(e.getLocalizedMessage());
513 		} catch (Exception e) {
514 			log.error( "While setting system environment to cli - " + e.getLocalizedMessage(), e);
515 			throw new MojoExecutionException(e.getLocalizedMessage());
516 		}
